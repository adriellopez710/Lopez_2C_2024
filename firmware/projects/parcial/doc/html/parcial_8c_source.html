162void iniciar(void) {
163 // Inicialización de periféricos
164 LedsInit();
165 SwitchesInit();
166 HcSr04Init(GPIO_3, GPIO_2);
167 BuzzerInit(2); // Inicializacion del buzzer

169 /* Configura el temporizador para el sensor de distancia */
170 timer_config_t timer_sensor = {
171 .timer = TIMER_A,
172 .period = CONFIG_SENSING_PERIOD_US_SENS, // Período de sensado de 100 Hz
173 .func_p = FuncTimerSensor,
174 .param_p = NULL
175 };
176 TimerInit(&timer_sensor);
177
178 /* Configura el temporizador para el acelerometro */
179 timer_config_t timer_acel = {
180 .timer = TIMER_B,
181 .period = CONFIG_SENSING_PERIOD_US_ACEL, // Período de sensado de 100 Hz
182 .func_p = FuncTimerAcel,
183 .param_p = NULL
184 };
185 TimerInit(&timer_acel);
186
187
188 /* Configuración del puerto serie */
189 serial_config_t ControlarUart =
190 {
191 .port = 2,
192 .baud_rate = 115200,
193 .func_p = NULL,
194 .param_p = NULL,
195 };
196 UartInit(&ControlarUart);
197
198
199 /* Creación de las tareas */
200 xTaskCreate(&SensorTask, "SensorTask", 2048, NULL, 5, &sensor_task_handle);
201 xTaskCreate(&LedTask, "LedTask", 2048, NULL, 4, &led_task_handle);
202 xTaskCreate(&AcelTask, "AcelTask", 2048, NULL, 6, &acel_task_handle);
203
204 /* Inicia de los timer */
205 TimerStart(timer_sensor.timer);
206 TimerStart(timer_acel.timer);
207
208}
